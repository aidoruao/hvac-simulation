     """
     FR-MA-001: Helmholtz Equation of State — First-Principles
   Implementation

     EOS-HEOS-001: Helmholtz residual form a(δ,τ) = a^ideal + a^res
     EOS-DER-001/002: First and second partial derivatives
     EOS-PROP-001: Derived thermodynamic properties
     """

     import numpy as np
     from typing import Tuple, Dict

     # R410A coefficients from Lemmon et al. (NIST REFPROP 10.0)
     # Reduced variables: δ = ρ/ρ_c, τ = T_c/T
     R410A_COEFFICIENTS = {
         "R": 8.314462618,  # J/(mol·K) — universal gas constant
         "M": 72.585,        # g/mol — molar mass
         "T_c": 344.494,     # K — critical temperature
         "rho_c": 459.030,   # kg/m³ — critical density
         "p_c": 4901.0,      # kPa — critical pressure

         # Ideal gas part coefficients (simplified — full set from REFPROP)
         "ideal": {
             "n": [np.log(1.0), 2.5],  # placeholder — full polynomial from
   Span 2000
             "t": [0.0, 1.0],
         },

         # Residual part coefficients (simplified — full set from REFPROP)
         "residual": {
             "n": [0.01, -0.05, 0.1, -0.02, 0.003],  # placeholder
             "d": [1, 1, 2, 3, 3],
             "t": [0.0, 0.5, 1.0, 1.5, 2.0],
             "l": [0, 0, 0, 1, 1],  # Gaussian exponent (0 = no exp term)
         }
     }


     class HelmholtzEOS:
         """
         Helmholtz Equation of State in reduced variables.

         Glass box: every method returns intermediate values for
   inspection.
         No hidden state. All derivatives computed analytically.
         """

         def __init__(self, fluid: str = "R410A"):
             self.fluid = fluid
             self.coeffs = R410A_COEFFICIENTS  # TODO: multi-fluid support
             self.R = self.coeffs["R"] / self.coeffs["M"] * 1000  #
   J/(kg·K)
             self.T_c = self.coeffs["T_c"]
             self.rho_c = self.coeffs["rho_c"]

         def _a_ideal(self, delta: float, tau: float) -> float:
             """Ideal gas contribution: a^ideal(δ,τ)"""
             n = self.coeffs["ideal"]["n"]
             t = self.coeffs["ideal"]["t"]
             return np.log(delta) + sum(nk * tau**tk for nk, tk in zip(n,
   t))

         def _a_res(self, delta: float, tau: float) -> float:
             """Residual contribution: a^res(δ,τ)"""
             n = self.coeffs["residual"]["n"]
             d = self.coeffs["residual"]["d"]
             t = self.coeffs["residual"]["t"]
             l = self.coeffs["residual"]["l"]

             a_res = 0.0
             for nk, dk, tk, lk in zip(n, d, t, l):
                 term = nk * delta**dk * tau**tk
                 if lk > 0:
                     term *= np.exp(-delta**lk)
                 a_res += term
             return a_res

         def a(self, delta: float, tau: float) -> float:
             """Total Helmholtz energy: a(δ,τ) = a^ideal + a^res"""
             return self._a_ideal(delta, tau) + self._a_res(delta, tau)

         def da_ddelta(self, delta: float, tau: float) -> float:
             """First partial: (∂a/∂δ)_τ"""
             # Ideal part derivative
             da_id = 1.0 / delta

             # Residual part derivative
             n = self.coeffs["residual"]["n"]
             d = self.coeffs["residual"]["d"]
             t = self.coeffs["residual"]["t"]
             l = self.coeffs["residual"]["l"]

             da_res = 0.0
             for nk, dk, tk, lk in zip(n, d, t, l):
                 if lk == 0:
                     da_res += nk * dk * delta**(dk - 1) * tau**tk
                 else:
                     # d/dδ [δ^d * exp(-δ^l)] = d*δ^(d-1)*exp(-δ^l) -
   l*δ^(d+l-1)*exp(-δ^l)
                     exp_term = np.exp(-delta**lk)
                     da_res += nk * tau**tk * exp_term * (dk * delta**(dk -
   1) - lk * delta**(dk + lk - 1))

             return da_id + da_res

         def d2a_ddelta2(self, delta: float, tau: float) -> float:
             """Second partial: (∂²a/∂δ²)_τ"""
             # TODO: implement analytical second derivative
             # Placeholder: complex step for verification
             h = 1e-100
             return (self.da_ddelta(delta + 1j*h, tau).imag / h).real

         def pressure(self, delta: float, tau: float) -> float:
             """EOS-PROP-001: P = ρRT[1 + δ(∂a^res/∂δ)_τ]"""
             T = self.T_c / tau
             rho = delta * self.rho_c
             da_res_ddelta = self.da_ddelta(delta, tau) - (1.0 / delta)  #
   subtract ideal part
             return rho * self.R * T * (1.0 + delta * da_res_ddelta)

         def solve_delta(self, P_target: float, T: float, delta_guess:
   float = 0.5) -> Tuple[float, Dict]:
             """
             EOS-CONV-001: Newton-Raphson for δ(P,T)

             Returns:
                 delta: converged reduced density
                 info: dict with iterations, residual history, Jacobian
   condition number
             """
             tau = self.T_c / T
             delta = delta_guess

             info = {"iterations": [], "residuals": [], "jacobians": []}

             for i in range(10):  # max 10 iterations
                 P_calc = self.pressure(delta, tau)
                 f = P_calc - P_target

                 # Analytical Jacobian: dP/dδ
                 # P = ρRT[1 + δ(∂a^res/∂δ)]
                 # dP/dδ = ρ_c * R * T * [1 + 2δ(∂a^res/∂δ) +
   δ²(∂²a^res/∂δ²)]
                 da_res = self.da_ddelta(delta, tau) - (1.0 / delta)
                 d2a_res = self.d2a_ddelta2(delta, tau) + (1.0 / delta**2)
                 J = self.rho_c * self.R * T * (1.0 + 2*delta*da_res +
   delta**2*d2a_res)

                 info["iterations"].append(i)
                 info["residuals"].append(abs(f))
                 info["jacobians"].append(abs(J))

                 if abs(f) < 1e-12:
                     info["converged"] = True
                     info["final_delta"] = delta
                     return delta, info

                 delta = delta - f / J

             info["converged"] = False
             info["final_delta"] = delta
             return delta, info


     if __name__ == "__main__":
         # Quick sanity check
         eos = HelmholtzEOS()
         delta, tau = 0.5, 1.0  # T = T_c, rho = 0.5 * rho_c
         print(f"a(δ={delta}, τ={tau}) = {eos.a(delta, tau):.6f}")
         print(f"∂a/∂δ = {eos.da_ddelta(delta, tau):.6f}")
         print(f"P = {eos.pressure(delta, tau):.2f} kPa")

     """
     FR-MA-001 Tests: Helmholtz EOS verification

     EOS-DER-001: First partial derivative check
     EOS-DER-002: Second partial derivative check
     EOS-CONV-001: Newton-Raphson convergence
     """

     import numpy as np
     import pytest
     from math_model.helmholtz_eos import HelmholtzEOS


     def test_derivative_complex_step():
         """Verify da/dδ against complex-step differentiation (Martins et
   al. 2003)."""
         eos = HelmholtzEOS()
         delta, tau = 0.5, 1.0

         # Complex step: f'(x) ≈ Im(f(x + ih)) / h, h = 1e-100
         h = 1e-100
         a_complex = eos.a(delta + 1j*h, tau)
         da_complex = a_complex.imag / h

         da_analytical = eos.da_ddelta(delta, tau)

         assert abs(da_analytical - da_complex) < 1e-14, \
             f"Derivative mismatch: analytical={da_analytical},
   complex_step={da_complex}"


     def test_pressure_round_trip():
         """EOS-PROP-001: P(δ,T) → solve_delta(P,T) → δ should recover
   original."""
         eos = HelmholtzEOS()
         delta_true = 0.3
         T = 300.0  # K
         tau = eos.T_c / T

         P = eos.pressure(delta_true, tau)
         delta_solved, info = eos.solve_delta(P, T, delta_guess=0.5)

         assert info["converged"], f"NR did not converge: {info}"
         assert abs(delta_solved - delta_true) < 1e-9, \
             f"Round-trip failed: true={delta_true}, solved={delta_solved}"


     def test_newton_convergence_limit():
         """EOS-CONV-001: Must converge within 10 iterations."""
         eos = HelmholtzEOS()
         T = 300.0
         delta_true = 0.5
         tau = eos.T_c / T
         P = eos.pressure(delta_true, tau)

         delta_solved, info = eos.solve_delta(P, T)

         assert info["converged"]
         assert len(info["iterations"]) <= 10, \
             f"Too many iterations: {len(info['iterations'])}"


     def test_critical_point_approach():
         """Near critical point: δ→1, τ→1 should not diverge."""
         eos = HelmholtzEOS()
         delta, tau = 0.99, 1.01  # Near critical

         # Should not raise
         P = eos.pressure(delta, tau)
         da = eos.da_ddelta(delta, tau)

         assert np.isfinite(P)
         assert np.isfinite(da)
